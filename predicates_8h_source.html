672  inline Sign orient_3d_inexact(
673  const double* p0, const double* p1,
674  const double* p2, const double* p3
675  ) {
676  double a11 = p1[0] - p0[0] ;
677  double a12 = p1[1] - p0[1] ;
678  double a13 = p1[2] - p0[2] ;
679 
680  double a21 = p2[0] - p0[0] ;
681  double a22 = p2[1] - p0[1] ;
682  double a23 = p2[2] - p0[2] ;
683 
684  double a31 = p3[0] - p0[0] ;
685  double a32 = p3[1] - p0[1] ;
686  double a33 = p3[2] - p0[2] ;
687 
688  double Delta = det3x3(
689  a11,a12,a13,
690  a21,a22,a23,
691  a31,a32,a33
692  );
693 
694  return geo_sgn(Delta);
695  }
